// ハンバーガーメニュー処理 function toggleHamburger() { const sidebar = document.getElementById('sidebar'); const overlay = document.getElementById('hamburgerMenu'); if (sidebar && overlay) { sidebar.classList.toggle('open'); overlay.style.display = sidebar.classList.contains('open') ? 'block' : 'none'; } } // ハンバーガーボタンのイベントリスナー設定 setTimeout(() => { const hamburgerBtn = document.getElementById('hamburgerBtn'); if (hamburgerBtn) { hamburgerBtn.addEventListener('click', toggleHamburger); } }, 50);
